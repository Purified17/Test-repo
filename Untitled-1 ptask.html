// For Loop Example 1: Printing numbers from 0 to 4 // This loop initializes a counter variable 'i' to 0. // It continues to execute as long as 'i' is less than 5. // After each iteration, 'i' is incremented by 1. for (let i = 0; i < 5; i++) { console.log(i); // Prints the current value of 'i' } // For Loop Example 2: Iterating through an array // This loop iterates over each element in the 'colors' array. // The loop variable 'i' represents the index of the current element. const colors = ["red", "green", "blue"]; for (let i = 0; i < colors.length; i++) { console.log(`Color at index ${i}: ${colors[i]}`); // Prints the index and the color } // While Loop Example 1: Counting down from 5 to 1 // This loop continues as long as the 'count' variable is greater than 0. // In each iteration, the current value of 'count' is printed, and then 'count' is decremented. let count = 5; while (count > 0) { console.log(count); // Prints the current value of 'count' count--; // Decrements 'count' by 1 } // While Loop Example 2: Repeating an action until a condition is met // This loop continues to prompt the user for input until they enter "yes". let answer = ""; while (answer !== "yes") { answer = prompt("Do you want to continue? (yes/no)"); // Prompts the user for input console.log(`You entered: ${answer}`); } // Do...While Loop Example 1: Printing a number at least once // This loop will execute the code inside the 'do' block at least once, // even if the condition in the 'while' part is initially false. let num = 0; do { console.log(num); // Prints the value of 'num' num++; // Increments 'num' by 1 } while (num < 3); // The loop continues as long as 'num' is less than 3 // Do...While Loop Example 2: Getting valid input from the user // This loop continues to prompt the user for a number until they enter a value greater than 10. let userInput; do { userInput = parseInt(prompt("Enter a number greater than 10:")); // Prompts for input and converts it to an integer console.log(`You entered: ${userInput}`); } while (userInput <= 10); // The loop continues as long as 'userInput' is less than or equal to 10
